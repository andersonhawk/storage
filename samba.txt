### samba share filesystem between linux/unix and windows ###

## add a samba user/pass ##
	$ smbpasswd -a thomas

## verify the samba user ##
	$ pdbedit -L

## smbclient usage ##
	$ smbclient -L localhost -U thomas

## restart smb service ##
	$ systemctl restart smbd.service
